CREATE TABLE Doctor (
    Doctor_id INT PRIMARY KEY,
    Specialization VARCHAR(100),
    Qualification VARCHAR(100),
    Name_First VARCHAR(50),
    Name_Middle VARCHAR(50),
    Name_Last VARCHAR(50),
    Salary INT
);

CREATE TABLE Patient (
    Patient_id INT PRIMARY KEY,
    Name_First VARCHAR(50),
    DOB DATE,
    Address_Locality VARCHAR(100),
    Address_City VARCHAR(100),
    Phone VARCHAR(20)
);

CREATE TABLE Medicine (
    Code VARCHAR(10) PRIMARY KEY,
    Price NUMERIC(10,2),
    Quantity INT
);

CREATE TABLE Doctor_Treats_Patient (
    Doctor_id INT,
    Patient_id INT,
    PRIMARY KEY (Doctor_id, Patient_id),
    CONSTRAINT fk_dtp_doctor
        FOREIGN KEY (Doctor_id)
        REFERENCES Doctor(Doctor_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    CONSTRAINT fk_dtp_patient
        FOREIGN KEY (Patient_id)
        REFERENCES Patient(Patient_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

CREATE TABLE Patient_Bills_Medicine (
    Patient_id INT,
    Medicine_Code VARCHAR(10),
    Quantity INT,
    PRIMARY KEY (Patient_id, Medicine_Code),
    CONSTRAINT fk_pbm_patient
        FOREIGN KEY (Patient_id)
        REFERENCES Patient(Patient_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    CONSTRAINT fk_pbm_medicine
        FOREIGN KEY (Medicine_Code)
        REFERENCES Medicine(Code)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

FROM Patient P

    ON P.Patient_id = DTP.Patient_id
WHERE DTP.Doctor_id = 1;

CREATE TABLE high_salary_doctors AS
SELECT *
FROM Doctor
WHERE Salary > 15000;

FROM Doctor D

    ON D.Doctor_id = DTP.Doctor_id;

-- 15) Display patient name and age calculated from DOB.
SELECT
    Name_First,
    AGE(DOB) AS full_age,
    EXTRACT(YEAR FROM AGE(DOB)) AS age_years
FROM Patient;


SELECT
    Name_First,
    UPPER(Name_First) AS upper_name,
    LOWER(Name_First) AS lower_name,
    INITCAP(LOWER(Name_First)) AS proper_name
FROM Patient;


SELECT
    Phone AS original_phone,
    TRIM(Phone) AS trimmed_phone,
    LTRIM(Phone) AS left_trimmed_phone,
    RTRIM(Phone) AS right_trimmed_phone
FROM Patient;


SELECT
    Patient_id,
    CONCAT(Name_First, ' - Phone: ', Phone) AS contact_info
FROM Patient;

SELECT
    Name_First,
    SUBSTRING(Name_First FROM 1 FOR 3) AS first_3_letters,
    POSITION('a' IN LOWER(Name_First)) AS position_of_a
FROM Patient;

SELECT
    Name_First AS original_name,
    REPLACE(Name_First, 'Ahmed', 'Ahmad') AS updated_name
FROM Doctor;

SELECT
    Name_First,
    Salary AS original_salary,
    CAST(Salary AS INTEGER) AS salary_as_integer,
    CAST(Salary AS TEXT) AS salary_as_text
FROM Doctor;
