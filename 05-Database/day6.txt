
WITH avg_salary_cte AS (
    SELECT
        dept_id,
        AVG(salary) AS avg_salary
    FROM Employee
    GROUP BY dept_id
)
SELECT
    d.dept_name,
    a.avg_salary
FROM avg_salary_cte a
JOIN Department d
    ON d.dept_id = a.dept_id
WHERE a.avg_salary > 4000;

DROP TABLE IF EXISTS Task;

CREATE TABLE Task (
    task_id SERIAL PRIMARY KEY,
    task_name VARCHAR(100),
    emp_id INT,
    priority VARCHAR(20),
    status VARCHAR(20),
    due_date DATE,
    FOREIGN KEY (emp_id) REFERENCES Employee(emp_id)
);

SELECT e.emp_name
FROM Employee e
WHERE NOT EXISTS (
    SELECT 1
    FROM Task t
    WHERE t.emp_id = e.emp_id
      AND t.status = 'Completed'
);


SELECT
    e.emp_name,
    COUNT(t.task_id) AS total_tasks
FROM Employee e
JOIN Task t
    ON e.emp_id = t.emp_id
GROUP BY e.emp_name
HAVING COUNT(t.task_id) > 2;
